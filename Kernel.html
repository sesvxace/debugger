def breakpoint
  klass = is_a?(Module) ? self : self.__class__
  line  = caller[0].split(':')[1].to_i
  breakpoints = SES::Debugger.breakpoints[klass] ||= []
  breakpoints << line unless breakpoints.include?(line)
end
